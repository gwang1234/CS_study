## 자료구조와 알고리즘 

- 자료구조: 일정규칙으로 정리된 자료
  - 서점에서 물건이나 책을 적절히 정리해야 필요할때 빨리 꺼내 쓸 수 있다.
- 알고리즘: 자료를 접근하는 규칙
- 선형구조 데이터가 일렬로 나열된 구조, 데이터가 앞뒤로 연결된 형태
  - 배열(Array), 연결 리스트(Linked List) 등이 대표적이다.

### 퀵정렬
```
public class Test {
    public static void run(int[] data)
    {
        quickSort(data, 0, data.length-1);

    }


    public static void quickSort (int[] arr, int left, int right) {
        if (left >= right) return;

        int pivot = potision(arr, left, right);

        quickSort(arr, left, pivot);
        quickSort(arr, pivot+1, right);
    }

    public static int potision(int[] arr, int ll, int rr) {

        int left = ll-1;
        int right = rr +1;
        int pivot = arr[(ll + rr) / 2];

        while (true) {

            do {
                left++;
            } while (arr[left] < pivot && left < right);

            do {
                right--;
            } while (arr[right] > pivot && left <= right);

            if (left >= right) {
                return right;
            }

            swap(arr, left, right);
        }
    }

    public static void swap(int[] arr, int left, int right) {
        int temp = arr[left];
        arr[left] = arr[right];
        arr[right] = temp;
    }

}
```

### 병합정렬
```
public class Test {
    public static void run(int[] data)
    {
        // 병합정렬: 안정적인 성능 구현
        mergeSort(data);

    }


    // 병합정렬
    public static int[] sorted;

    public static void mergeSort(int[] data) {
        sorted = new int[data.length];
        merge(data, 0, data.length-1);
        sorted = null;
    }

    public static void  merge(int[] data, int left, int right) {
        if (left == right) return;

        int mid = (left+right) /2;
        merge(data, left, mid);
        merge(data, mid+1, right);
        merge(data, left, mid, right);
    }

    public static void merge(int[] data, int left, int mid, int right) {

        int index = left;
        int r = mid + 1;
        int l = left;

        while (l <= mid && r <= right) {
            if (data[l] <= data[r]) {
                sorted[index] = data[l];
                l++;
                index++;
            } else {
                sorted[index] = data[r];
                r++;
                index++;
            }
        }

        if (l > mid) {
            while ( r <= right) {
                sorted[index] = data[r];
                r++;
                index++;
            }
        } else {
            while (l <= mid) {
                sorted[index] = data[l];
                l++;
                index++;
            }
        }

        for (int i = left; i<=right; i++) {
            data[i] = sorted[i];
        }
    }

}

    E push(E item);
    E pop(  );
    E peek();
    int search(Object value);
    int size();
    void clear();
    boolean empty();
}
```

```
import java.util.Arrays;
import java.util.EmptyStackException;

public class Main<E> implements StackInterface<E>
{

    private static final int DEFAULT_CAPACITY = 10;
    private static final Object[] EMPTY_ARRAY = {};
    private Object[] array;
    private int size;

    public Main() {
        this.array = EMPTY_ARRAY;
        size = 0;
    }

    public Main(int capacity) {
        this.array = new Object[capacity];
        size = 0;
    }

    public void resize() {
        if (Arrays.equals(array, EMPTY_ARRAY)) {
            array = new Object[DEFAULT_CAPACITY];
            return;
        }

        int arrayCapacity = array.length;

        if (size == arrayCapacity) {
            arrayCapacity = arrayCapacity*2;
            array = Arrays.copyOf(array, arrayCapacity);
            return;
        }

        if (size < arrayCapacity/2) {
            arrayCapacity = arrayCapacity/2;
            array = Arrays.copyOf(array, Math.max(arrayCapacity, DEFAULT_CAPACITY));
            return;
        }
    }

    @Override
    public E push(E item) {

        if (size == array.length) {
            resize();
        }

        array[size-1] = item;
        size++;
        return item;
    }

    @Override
    public E pop() {

        if (size == 0) {
            throw new EmptyStackException();
        }

        E obj = (E) array[size-1];
        array[size-1] = null;
        size--;

        return obj;
    }

    @Override
    public E peek() {

        if (size == 0) {
            throw new EmptyStackException();
        }
        return (E) array[size-1];
    }

    @Override
    public int search(Object value) {

        if (value == null) {
            for (int i = size-1; i >=0; i--) {
                if (array[i] == null) {
                    return size-i;
                }
            }
        } else {
            for (int i = size-1; i >=0; i--) {
                if (array[i] == value) {
                    return size-i;
                }
            }
        }

        return -1;
    }

    @Override
    public int size() {
        return size;
    }

    @Override
    public void clear() {
        for (int i = 0; i < size; i++) {
            array[i] = null;
        }
        size = 0;
        resize();
    }

    @Override
    public boolean empty() {
        return size == 0;
    }

}
```

public interface QueueInterface<E> {
    boolean offer(E e);
    E poll();
    E peek();
}
```

```
import java.util.NoSuchElementException;

public class Main<E> implements QueueInterface<E>
{
    private static final int DEFAULT_ARRAY = 64;
    private Object[] array;
    private int size;
    private int front;
    private int rear;

    public Main() {
        this.array = new Object[DEFAULT_ARRAY];
        this.size = 0;
        this.front = 0;
        this. rear = 0;
    }

    public Main(int capacity) {
        this.array = new Object[capacity];
        this.size = 0;
        this.front = 0;
        this. rear = 0;
    }

    private void resize(int capacity) {
        int arrayCapacity = array.length;
        Object[] newarray = new Object[capacity];

        for (int i = 1, j = front+1; i <= size; i++, j++) {
            newarray[i] = array[j % arrayCapacity];
        }

        this.array = null;
        this.array = newarray;

        front = 0;
        rear = size;
    }

    @Override
    public boolean offer(E e) {

        if ((rear + 1) % array.length == front) {
            resize(array.length * 2);
        }

        rear = (rear+1) % array.length;
        array[rear] = e;
        size++;

        return true;
    }

    @Override
    public E poll() {

        if (size == 0) {
            return null;
        }

        front = (front+1) % array.length;

        @SuppressWarnings("unchecked")
        E obj = (E) array[front];
        array[front] = null;
        size--;

        if (array.length > DEFAULT_ARRAY && size < (array.length/4)) {
            resize(Math.max(DEFAULT_ARRAY, array.length / 2));
        }

        return obj;
    }

    public E remove() {

        E item = poll();

        if (item == null) {
            throw new NoSuchElementException();
        }

        return item;
    }

    @Override
    public E peek() {

        if (size == 0) {
            return  null;
        }

        @SuppressWarnings("unchecked")
        E obj = (E) array[(front+1) % array.length];

        return obj;
    }

    public E element() {

        E item = peek();

        if (item == null) {
            throw new NoSuchElementException();
        }

        return item;
    }

    public int size() {
        return size;
    }

    public boolean isEmpty() {
        return size == 0;
    }

    public boolean contains(Object value) {

        int start = (front + 1) % array.length;

        for (int i = 0, j = start; i < size; i++, j = (j+1) % array.length) {
            if (array[j].equals(value)) {
                return true;
            }
        }

        return false;
    }

    public void clear() {
        for (int i = 0; i< array.length; i++) {
            array[i] = null;
        }

        front = rear = size = 0;
    }
}
